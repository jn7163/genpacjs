; 配置

[config]
; 获取GFWList的代理设置，与PAC内的配置可能不同
; 如果你可以正常访问gfwlist，可以设置为不使用代理
; ProxyType 0 不使用代理; 1 SOCKS4; 2 SOCKS5; 3 HTTP
gfwProxyType = 0
gfwProxyHost = 127.0.0.1
gfwProxyPort = 1080
;gfwProxyUsr  = 
;gfwProxyPwd  = 

; gfwlist地址
gfwUrl = https://raw.githubusercontent.com/gfwlist/gfwlist/master/gfwlist.txt

; PAC的代理配置
; 注意：如果是在MAC下的SOCKS代理，必须设置成SOCKS5
; proxyType 1 SOCKS e.g. SOCKS 127.0.0.1:9527
;           2 SOCKS5 e.g. SOCKS5 127.0.0.1:9527; SOCKS 127.0.0.1:9527
;           3 HTTP e.g. PROXY 127.0.0.1:9527
; 默认 SOCKS5
pacProxyType = 2
pacProxyHost = 127.0.0.1
pacProxyPort = 1080

; 生成的PAC文件名 默认 AutoProxy.pac
pacFilename = AutoProxy.txt

; 调试模式
; 调试模式会输出更多的信息，且会额外产生如下文件
;   tmp/gfwlist.txt 经过base64解码后的gfwlist
;   tmp/rules.txt gfwlist、user-rules解析前后各条规则的对比
DebugMode = true

; JS版本
; 生成test/genpac.js 内容与PAC完全一样，只是改了文件名用于网址测试
JSVersion = false
